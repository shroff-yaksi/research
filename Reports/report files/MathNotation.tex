\begin{center}
{\LARGE\bf LIST OF MATHEMATICAL NOTATIONS}
\end{center}
\addcontentsline{toc}{chapter}{LIST OF MATHEMATICAL NOTATIONS}
\vspace{0.5cm}

\noindent $\mathcal{D}$ \dotfill Training dataset, including all rows and their class labels \\[4pt]
\noindent $\mathbf{x}$ \dotfill A single data row (feature vector) \\[4pt]
\noindent $y \in \{0, 1\}$ \dotfill Class label (1 = minority event, 0 = majority) \\[4pt]
\noindent $N$ \dotfill Total number of rows in the dataset \\[4pt]
\noindent $N_1$ \dotfill Count of minority class rows \\[4pt]
\noindent $\rho = N_1/N$ \dotfill Minority ratio, the fraction of minority rows \\[4pt]
\noindent $\pi = P(y=1)$ \dotfill Target minority ratio specified at generation time \\[4pt]
\noindent $\mathcal{X}$ \dotfill Feature space \\[4pt]
\noindent $d_{num}$, $d_{cat}$ \dotfill Number of numerical and categorical features \\[4pt]
\noindent $\mathcal{C}_j$ \dotfill Categorical domain for feature $j$ \\[4pt]
\noindent $G_\theta$ \dotfill Generative model parameterised by $\theta$ \\[4pt]
\noindent $G$, $D$ \dotfill Generator and discriminator networks in a GAN \\[4pt]
\noindent $\mathcal{L}_{GAN}$ \dotfill GAN minimax objective \\[4pt]
\noindent $p_{data}(\mathbf{x})$ \dotfill True data distribution \\[4pt]
\noindent $p_\theta(\mathbf{x})$ \dotfill Model's learned distribution \\[4pt]
\noindent $q_\phi(\mathbf{z}|\mathbf{x})$ \dotfill VAE encoder distribution parameterised by $\phi$ \\[4pt]
\noindent $p_\theta(\mathbf{x}|\mathbf{z})$ \dotfill VAE decoder distribution parameterised by $\theta$ \\[4pt]
\noindent $\phi$ \dotfill Encoder network parameters \\[4pt]
\noindent $\mathbf{z}$ \dotfill Latent vector produced by the VAE encoder \\[4pt]
\noindent $d_z = 64$ \dotfill Dimensionality of the latent space \\[4pt]
\noindent $\boldsymbol{\mu}$, $\boldsymbol{\sigma}^2$ \dotfill Mean and variance output by the VAE encoder \\[4pt]
\noindent $\boldsymbol{\epsilon}$ \dotfill Random Gaussian noise (reparameterisation or diffusion) \\[4pt]
\noindent $\beta_{\text{VAE}}$ \dotfill KL weight in the VAE loss (set to 0.1) \\[4pt]
\noindent $t$ \dotfill Diffusion timestep \\[4pt]
\noindent $\mathbf{z}_t$ \dotfill Noisy latent vector at timestep $t$ \\[4pt]
\noindent $T = 1000$ \dotfill Total number of diffusion timesteps \\[4pt]
\noindent $\beta_t$ \dotfill Noise variance added at timestep $t$ (linearly increasing) \\[4pt]
\noindent $\alpha_t = 1 - \beta_t$ \dotfill Signal retention at timestep $t$ \\[4pt]
\noindent $\bar{\alpha}_t$ \dotfill Cumulative signal retention factor at timestep $t$ \\[4pt]
\noindent $\epsilon_\theta$ \dotfill DiT noise prediction network parameterised by $\theta$ \\[4pt]
\noindent $w$ \dotfill CFG guidance scale (default 2.0) \\[4pt]
\noindent $p_{uncond}$ \dotfill Label dropout probability during CFG training (0.10) \\[4pt]
\noindent $\emptyset$ \dotfill Null token used when class label is dropped \\[4pt]
\noindent $\tilde{\epsilon}$ \dotfill Guided noise prediction (CFG output) \\[4pt]
\noindent $\mathcal{L}_{VAE}$ \dotfill Total VAE training loss \\[4pt]
\noindent $\mathcal{L}_{recon}$ \dotfill Reconstruction loss component of VAE \\[4pt]
\noindent $D_{KL}$ \dotfill Kullback-Leibler divergence \\[4pt]
\noindent $\text{KS}$ \dotfill Kolmogorov-Smirnov statistic (distributional fidelity) \\[4pt]
\noindent $\text{F1}_{min}$ \dotfill F1-score for the minority class (primary utility metric) \\[4pt]
\noindent $\text{AUC}$ \dotfill Area Under the ROC Curve (overall utility metric) \\[4pt]
\noindent $\text{DCR}$ \dotfill Distance to Closest Record (empirical privacy metric)
